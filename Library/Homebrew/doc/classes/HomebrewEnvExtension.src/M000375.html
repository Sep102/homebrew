  def llvm
    xcode_path = `/usr/bin/xcode-select -print-path`.chomp
    xcode_path = "/Developer" if xcode_path.to_s.empty?
    self['CC'] = "#{xcode_path}/usr/bin/llvm-gcc"
    self['CXX'] = "#{xcode_path}/usr/bin/llvm-g++"
    self['LD'] = self['CC']
    self.O4
  end
