  def test_p1_patch
    nostdout do
      P1PatchBall.new('test_p1_patch').brew do
        s = read_file 'libexec/NOOP'
        assert !s.include?("NOOP"), "File was unpatched."
        assert s.include?("ABCD"), "File was not patched as expected."
      end
    end
  end
