def check_for_blacklisted_formula names
  return if ARGV.force?

  names.each do |name|
    case name
    when 'tex', 'tex-live', 'texlive' then abort "Installing TeX from source is weird and gross, requires a lot of patches,\nand only builds 32-bit (and thus can't use Homebrew deps on Snow Leopard.)\n\nWe recommend using a MacTeX distribution:\nhttp://www.tug.org/mactex/\n".undent

    when 'mercurial', 'hg' then abort "Mercurial can be install thusly:\nbrew install pip && pip install mercurial\n".undent

    when 'setuptools' then abort "When working with a Homebrew-built Python, distribute is preferred\nover setuptools, and can be used as the prequisite for pip.\n\nInstall distribute using:\nbrew install distribute\n".undent
    end
  end
end
