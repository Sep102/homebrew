def __make url, name
  require 'formula'
  require 'digest'
  require 'erb'

  path = Formula.path(name)
  raise "#{path} already exists" if path.exist?

  if Formula.aliases.include? name and not ARGV.force?
    realname = HOMEBREW_REPOSITORY.join("Library/Aliases/#{name}").realpath.basename('.rb')
    raise "The formula \#{realname} is already aliased to \#{name}\nPlease check that you are not creating a duplicate.\nTo force creation use --force.\n".undent
  end

  if ARGV.include? '--cmake'
    mode = :cmake
  elsif ARGV.include? '--autotools'
    mode = :autotools
  else
    mode = nil
  end

  version = Pathname.new(url).version
  if version.nil?
    opoo "Version cannot be determined from URL."
    puts "You'll need to add an explicit 'version' to the formula."
  else
    puts "Version detected as #{version}."
  end

  md5 = ''
  if ARGV.include? "--cache" and version != nil
    strategy = detect_download_strategy url
    if strategy == CurlDownloadStrategy
      d = strategy.new url, name, version, nil
      the_tarball = d.fetch
      md5 = the_tarball.md5
      puts "MD5 is #{md5}"
    else
      puts "--cache requested, but we can only cache formulas that use Curl."
    end
  end

  formula_template = "require 'formula'\n\nclass \#{Formula.class_s name} <Formula\nurl '\#{url}'\nhomepage ''\nmd5 '\#{md5}'\n\n<% if mode == :cmake %>\ndepends_on 'cmake'\n<% elsif mode == nil %>\n# depends_on 'cmake'\n<% end %>\n\ndef install\n<% if mode == :cmake %>\nsystem \"cmake . \\\#{std_cmake_parameters}\"\n<% elsif mode == :autotools %>\nsystem \"./configure\", \"--prefix=\\\#{prefix}\", \"--disable-debug\", \"--disable-dependency-tracking\"\n<% else %>\nsystem \"./configure\", \"--prefix=\\\#{prefix}\", \"--disable-debug\", \"--disable-dependency-tracking\"\n# system \"cmake . \\\#{std_cmake_parameters}\"\n<% end %>\nsystem \"make install\"\nend\nend\n"

  path.write(ERB.new(formula_template, nil, '>').result(binding))
  return path
end
