def text_for_keg_only_formula f
  if f.keg_only? == :provided_by_osx
    rationale = "This is because the formula is already provided by OS X."
  elsif f.keg_only?.kind_of? String
    rationale = "The formula provides the following rationale:\n\n#{f.keg_only?.chomp}"
  else
    rationale = "The formula didn't provide any rationale for this."
  end
  "\#{f.name} is keg-only. This means it is not symlinked into Homebrew's\nprefix. \#{rationale}\n\nGenerally there are no consequences of this for you, however if you build your\nown software and it requires this formula, you may want to run this command to\nlink it into the Homebrew prefix:\n\nbrew link \#{f.name}\n"
end
