def check_for_symlinked_cellar
  if HOMEBREW_CELLAR.symlink?
    puts "Symlinked Cellars can cause problems.\nYour Homebrew Cellar is a symlink: \#{HOMEBREW_CELLAR}\nwhich resolves to: \#{HOMEBREW_CELLAR.realpath}\n\nThe recommended Homebrew installations are either:\n(A) Have Cellar be a real folder inside of your HOMEBREW_PREFIX\n(B) Symlink \"bin/brew\" into your prefix, but don't symlink \"Cellar\".\n\nOlder installations of Homebrew may have created a symlinked Cellar, but this can\ncause problems when two formula install to locations that are mapped on top of each\nother during the linking step.\n\n".undent
  end
end
