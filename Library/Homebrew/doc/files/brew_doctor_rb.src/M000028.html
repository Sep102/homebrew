def check_gcc_versions
  gcc_42 = gcc_42_build
  gcc_40 = gcc_40_build

  if gcc_42 == nil
    puts "We couldn't detect gcc 4.2.x. Some formulas require this compiler.\n\n".undent
  elsif gcc_42 < RECOMMENDED_GCC_42
    puts "Your gcc 4.2.x version is older than the recommended version. It may be advisable\nto upgrade to the latest release of Xcode.\n\n".undent
  end

  if gcc_40 == nil
    puts "We couldn't detect gcc 4.0.x. Some formulas require this compiler.\n\n".undent
  elsif gcc_40 < RECOMMENDED_GCC_40
    puts "Your gcc 4.0.x version is older than the recommended version. It may be advisable\nto upgrade to the latest release of Xcode.\n\n".undent
  end
end
