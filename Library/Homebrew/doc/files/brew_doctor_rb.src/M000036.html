def check_which_pkg_config
  binary = `/usr/bin/which pkg-config`.chomp
  return if binary.empty?

  unless binary == "#{HOMEBREW_PREFIX}/bin/pkg-config"
    puts "You have a non-brew 'pkg-config' in your PATH:\n\#{binary}\n\n`./configure` may have problems finding brew-installed packages using\nthis other pkg-config.\n\n".undent
  end
end
