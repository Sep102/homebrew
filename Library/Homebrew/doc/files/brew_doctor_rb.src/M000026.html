def check_for_nonstandard_x11
  return unless File.exists? '/usr/X11'
  x11 = Pathname.new('/usr/X11')
  if x11.symlink?
    puts "\"/usr/X11\" was found, but it is a symlink to:\n\#{x11.resolved_path}\n\nHomebrew's X11 support has only be tested with Apple's X11.\nIn particular, \"XQuartz\" and \"XDarwin\" are not known to be compatible.\n\n".undent
  end
end
