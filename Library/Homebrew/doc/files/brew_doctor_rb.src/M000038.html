def check_for_gettext
  if File.exist? "#{HOMEBREW_PREFIX}/lib/libgettextlib.dylib" or
     File.exist? "#{HOMEBREW_PREFIX}/lib/libintl.dylib"
    puts "gettext was detected in your PREFIX.\n\nThe gettext provided by Homebrew is \"keg-only\", meaning it does not\nget linked into your PREFIX by default.\n\nIf you `brew link gettext` then a large number of brews that don't\notherwise have a `depends_on 'gettext'` will pick up gettext anyway\nduring the `./configure` step.\n\n".undent
  end
end
