def check_usr_bin_ruby
  if /^1\.9/.match RUBY_VERSION
    puts "Ruby version \#{RUBY_VERSION} is unsupported.\nHomebrew is developed and tested on Ruby 1.8.x, and may not work correctly\non Ruby 1.9.x. Patches are accepted as long as they don't break on 1.8.x.\n\n".undent
  end
end
