def check_for_other_package_managers
  if macports_or_fink_installed?
    puts "You have Macports or Fink installed. This can cause trouble.\nYou don't have to uninstall them, but you may like to try temporarily\nmoving them away, eg.\n\nsudo mv /opt/local ~/macports\n\n".undent
  end
end
