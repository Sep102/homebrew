def check_for_git
  git = `/usr/bin/which git`.chomp
  if git.empty?
    puts "\"Git\" was not found in your path.\n\nHomebrew uses Git for several internal functions, and some formulae\n(Erlang in particular) use Git checkouts instead of stable tarballs.\n\nYou may want to do:\nbrew install git\n\n".undent
  end
end
