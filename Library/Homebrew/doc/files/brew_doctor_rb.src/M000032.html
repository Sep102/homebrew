def check_access_pkgconfig
  # If PREFIX/lib/pkgconfig already exists, "sudo make install" of
  # non-brew installed software may cause installation failures.
  pkgconfig = HOMEBREW_PREFIX+'lib/pkgconfig'
  return unless pkgconfig.exist?

  unless pkgconfig.writable?
    puts "\#{pkgconfig} isn't writable.\nThis can happen if you \"sudo make install\" software that isn't managed\nby Homebrew. If a brew tries to write a .pc file to this folder, the\ninstall will fail during the link step.\n\nYou should probably `chown` \#{pkgconfig}\n\n".undent
  end
end
